22def load_websites(): 

23 """Load the latest website data from route53 and s3 into the database.""" 

24 initial_load = route53.list_hosted_zones().get("HostedZones") 

25 

26 domain_load = [] 

27 for zone in initial_load: 

28 hosted_zone = { 

29 "name": zone["Name"][:-1], 

30 "route53": {"id": zone["Id"]}, 

31 } 

32 hosted_zone["is_active"] = False 

33 domain_load.append(hosted_zone) 

34 

35 # list websites within the S3 Repository 

36 websites = list_top_level_prefixes(WEBSITE_BUCKET) 

37 

38 # load available websites if available 

39 data_load = [] 

40 for domain in domain_load: 

41 if not website_manager.get(filter_data={"name": domain.get("name")}): 

42 domain_name = domain.get("name") 

43 if domain_name in websites: 

44 domain["s3_url"] = f"{WEBSITE_BUCKET_URL}/{domain_name}" 

45 data_load.append(domain) 

46 

47 # save latest data to the database 

48 if data_load != []: 

49 website_manager.save_many(data=data_load) 

50 logger.info("Database has been synchronized with domain data.")
