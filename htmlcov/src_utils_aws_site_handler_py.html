230def generate_ssl_certs(website): 

231 """Request and Validate an SSL certificate using AWS Certificate Manager.""" 

232 domain_name = website["name"] 

233 dns_id = website["route53"]["id"] 

234 requested_certificate = acm.request_certificate( 

235 DomainName=domain_name, 

236 ValidationMethod="DNS", 

237 SubjectAlternativeNames=[domain_name], 

238 DomainValidationOptions=[ 

239 {"DomainName": domain_name, "ValidationDomain": domain_name}, 

240 ], 

241 Options={"CertificateTransparencyLoggingPreference": "ENABLED"}, 

242 ) 

243 

244 certificate_arn = requested_certificate["CertificateArn"] 

245 resource_records = {} 

246 while not resource_records: 

247 time.sleep(2) 

248 certificate_description = acm.describe_certificate( 

249 CertificateArn=certificate_arn 

250 ) 

251 resource_records = [ 

252 description.get("ResourceRecord", None) 

253 for description in certificate_description.get("Certificate", {}).get( 

254 "DomainValidationOptions" 

255 ) 

256 ][0] 

257 

258 # add validation record to the dns 

259 route53.change_resource_record_sets( 

260 HostedZoneId=dns_id, 

261 ChangeBatch={ 

262 "Comment": domain_name, 

263 "Changes": [ 

264 { 

265 "Action": "UPSERT", 

266 "ResourceRecordSet": { 

267 "Name": resource_records["Name"], 

268 "Type": "CNAME", 

269 "TTL": 30, 

270 "ResourceRecords": [{"Value": resource_records["Value"]}], 

271 }, 

272 } 

273 ], 

274 }, 

275 ) 

276 

277 # wait until the certificate has been validated 

278 waiter = acm.get_waiter("certificate_validated") 

279 waiter.wait(CertificateArn=certificate_arn) 

280 

281 return certificate_arn
