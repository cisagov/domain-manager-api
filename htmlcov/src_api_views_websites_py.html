361class WebsiteCategorizeView(MethodView): 

362 """WebsiteCategorizeView.""" 

363 

364 def get(self, website_id): 

365 """Manage categorization of active sites.""" 

366 browserless_endpoint = os.environ.get("BROWSERLESS_ENDPOINT") 

367 chrome_options = webdriver.ChromeOptions() 

368 chrome_options.add_argument("--no-sandbox") 

369 chrome_options.add_argument("--headless") 

370 website = website_manager.get(document_id=website_id) 

371 domain = website["name"] 

372 if website.get("is_categorized", None): 

373 return {"error": f"{domain} has already been categorized."} 

374 

375 category = category_manager.get( 

376 filter_data={"name": request.args.get("category", "").capitalize()} 

377 ) 

378 

379 if not category: 

380 return {"error": "Category does not exist"} 

381 

382 is_category_submitted = [] 

383 # Submit domain to proxy 

384 if not current_app.config["TESTING"]: 

385 proxies = proxy_manager.all() 

386 for proxy in proxies: 

387 proxy_name = proxy["name"] 

388 

389 # Get unique category name for each proxy 

390 proxy_category = "".join( 

391 detail.get(proxy_name) 

392 for detail in category.get("proxies") 

393 if proxy_name in detail 

394 ) 

395 

396 try: 

397 driver = webdriver.Remote( 

398 command_executor=f"http://{browserless_endpoint}/webdriver", 

399 desired_capabilities=chrome_options.to_capabilities(), 

400 ) 

401 driver.set_page_load_timeout(60) 

402 exec( 

403 proxy.get("script"), 

404 { 

405 "driver": driver, 

406 "url": proxy.get("url"), 

407 "domain": domain, 

408 "api_key": two_captcha_api_key, 

409 "category": proxy_category, 

410 }, 

411 ) 

412 driver.quit() 

413 is_category_submitted.append( 

414 { 

415 "_id": proxy["_id"], 

416 "name": proxy_name, 

417 "is_categorized": False, 

418 } 

419 ) 

420 logging.info(f"Categorized with {proxy_name}") 

421 except Exception as err: 

422 driver.quit() 

423 logging.error(f"{proxy_name} has failed: {str(err)}") 

424 

425 # Quit WebDriver 

426 driver.quit() 

427 

428 # Update database 

429 website_manager.update( 

430 document_id=website_id, 

431 data={"is_category_submitted": is_category_submitted}, 

432 ) 

433 return jsonify( 

434 {"message": f"{domain} has been successfully submitted for categorization"}
